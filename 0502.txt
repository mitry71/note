package javaClass;
//리모콘 인터페이스
public interface RemoteControl {
	//상수; 인터페이스는 일반적인 필드는 가질 수 없다
	int MAX_VOLUME=10;  //public static final 생략 가능
	int MIN_VOLUME=0;
	
	//추상메소드
	void turnOn();  //abstract 생략 가능; 인터페이스는 일반메소드(실체 존재)를 가질 수 없기 때문에
	void turnOff();
	void setVolume(int volume);
	
	//디폴트메소드
	default void setMute(boolean mute) {
		if(mute) {
			System.out.println("무음 처리합니다.");
		}else {
			System.out.println("무음 해제합니다.");
		}
	}
	
	//정적메소드: 인스턴스 객체를 생성하지 않고 사용하는 메소드
	static void changeBattery() {
		System.out.println("건전지를 교환합니다.");	
	}
}

package javaClass;

public class Television implements RemoteControl {
	//필드
	private int volume;
	
	@Override
	public void turnOn() {
		System.out.println("TV를 켭니다.");		
	}

	@Override
	public void turnOff() {
		System.out.println("TV를 끕니다.");
	}

	@Override
	public void setVolume(int volume) {		
		if(volume > RemoteControl.MAX_VOLUME) {
			this.volume = RemoteControl.MAX_VOLUME;
		}else if(volume < RemoteControl.MIN_VOLUME) {
			this.volume = RemoteControl.MIN_VOLUME;
		}else {
			this.volume = volume;
		}
		System.out.println("현재 TV 볼륨 : " + this.volume);
	}
}

public class AClassEx {

	public static void main(String[] args) {
		AClass aa=new AClass(); //인스턴스 객체 생성

		bb.bField=10;
		bb.bMethod();
	}

}

내부 클래스를 사용하려면 외부 클래스의 인스턴스 객체를 먼저 생성하여 사용해야 한다.
