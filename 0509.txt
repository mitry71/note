package javaAPI;

public class SystemTimeEx1 {

	public static void main(String[] args) {
		//long time1=System.currentTimeMillis(); 천분의 1초
		long time1=System.nanoTime(); //10억분의 1초; 시스템으로부터 현재 시간을 나노초로 읽어옴
		int sum=0;
		for(int i=1; i<=1000000; i++) {
			sum+=i;
		}
		long time2=System.nanoTime();
		System.out.println("1부터 1000000까지의 합=" + sum);
		System.out.println("계산작업에 " + (time2-time1) + "나노초가 소요됨");
	}

}


package javaAPI;

import java.util.Scanner;
import java.util.regex.Pattern;

public class PatternEx1 {

	public static void main(String[] args) {
		//정규표현식
		Scanner sc=new Scanner(System.in);
		String regExp="(02|010)-\\d{3,4}-\\d{4}";  //앞자리는 010, - 나와야 함, \는 이스케이프 시퀀스로 \d를 읽게, {3,4}는 3에서 4개까지
		System.out.print("전화번호를 입력>>");
		String data=sc.next();
		boolean result=Pattern.matches(regExp, data);
		if(result) {
			System.out.println("전화번호 형식이 일치합니다.");
		}else {
			System.out.println("전화번호 형식과 틀립니다.");
		}
		System.out.print("이메일을 입력>>");
		String email=sc.next();
		String regExpEmail="\\w+@\\w+\\.\\w+(\\.\\w+)?";	 //.은 그냥 못씀, \\.; 그냥 .은 한 개의 문자(영문자, 한글문자, 특수문자 등)를 뜻함
		result=Pattern.matches(regExpEmail, email);
		if(result) {
			System.out.println("이메일주소 형식입니다.");
		}else {
			System.out.println("이메일주소 형식이 아닙니다.");
		}
	}

}


package javaAPI;

import java.util.Arrays;

public class ArraysEx1 {

	public static void main(String[] args) {
		int[] scores= {75,96,80,90,83};
		Arrays.sort(scores);  //소트 메소드(오름차순)
		for(int i=scores.length-1; i>=0; i--) {  //내림차순으로 출력
			System.out.println(scores[i]);
		}
		String[] names= {"박철순","홍길동","이영희"};
		Arrays.sort(names);  //문자열 소트(오름차순)
		for(int i=0; i<names.length; i++) {
			System.out.println(names[i]);
		}
	}

}

		System.out.println("찾은 위치=" + index);

		Arrays.sort(names);
		index=Arrays.binarySearch(names, "홍길동");
		System.out.println("찾은 위치=" + index);
	}

}

package javaAPI;

public class MathEx1 {

	public static void main(String[] args) {
		int num1=Math.abs(-5);  //절댓값
		System.out.println(num1);
		double num2=Math.ceil(3.2);  //올림
		System.out.println(num2);
		double num3=Math.round(3.2);  //반올림
		System.out.println(num3);
		double num4=Math.floor(3.2);  //내림
		System.out.println(num4);
		double num5=Math.rint(5.3);  //가까운 정수의 실수값
		System.out.println(num5);
		System.out.println(Math.random());  //0~1 난수를 발생
		System.out.println(Math.max(15, 50));
		System.out.println(Math.min(2.3, 45.7));
	}

}


package javaAPI;

import java.util.Arrays;
import java.util.Random;

public class RandomEx1 {

	public static void main(String[] args) {
		//Random 클래스
		int myNum[]=new int[6];
		Random rand1=new Random(3);  //같은 시드 번호를 주면 그 시드 번호에 해당하는 알고리즘에 따라 같은 난수 발생
		System.out.println("내가 선택한 번호 : ");
		for(int i=0; i<6; i++) {
			myNum[i]=rand1.nextInt(45)+1;
			System.out.print(myNum[i] + " ");
		}
		System.out.println();
		int lottoNum[]=new int[6];
		Random rand2=new Random(5);
		for(int i=0; i<6; i++) {
			lottoNum[i]=rand2.nextInt(45)+1;
			System.out.print(lottoNum[i] + " ");
		}
		System.out.println();
		Arrays.sort(myNum);
		Arrays.sort(lottoNum);
		boolean result=Arrays.equals(myNum, lottoNum);
		if(result) {
			System.out.println("1등에 당첨되셨습니다.");
		}else {
			System.out.println("아쉽네요");
		}
	}

}
