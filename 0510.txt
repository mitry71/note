<!DOCTYPE html>  <!--문서의 형식 선언:웹표준 HTML5-->
<html lang="ko">  <!--웹문서 시작 / 웹 접근성-->
<head>  <!--설정-->
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <link rel="shortcut icon" href="images/girl.ico">  <!--파비콘-->
    <title>SB</title>
</head>
<body>  <!--실제 페이지가 보이는 곳-->
    <h1>오늘의 명언</h1>  <!--홈페이지 제목-->
    <h2>시간이란...</h2>
    <img src="images/watch.jpg" />
    <p>내일 죽을 것처럼 오늘을 살고 영원히 살 것처럼 내일을 꿈꾸어라</p>
</body>
</html>
